import random

import random

tp = 0

def ask_again():
    ans = input("Do you want to play again? (y/n): ")
    return ans

def choose_game():
    user_choice = int(input("What game do you want to play? (1: heads or tails, 2: slot machine, 3: lucky seven, 4: blackjack, 5: roulette: "))
    return user_choice

def heads_or_tails():
    print("The rules of this game are simple, choose heads or tails, if you are correct, you win!")
    points = 0
    while True:
        choice = int(input("Enter 1 or 2 (1 is heads, 2 is tails): "))
        compchoice = random.randint(1, 2)
        print(f"The result was: {compchoice}")
        if choice == compchoice:
            print("You win!!")
            points = points + 1
        else:
            print("You lose!")
        user_ans = ask_again()
        if user_ans == "n":
            break
    
    return points

def slot_machine():
    print("The player spins three symbols and wins if they match.")
    points = 0
    while True:
        symbols = ["Cherries", "Lemon", "Bar", "Star", "Cash"]
        compchoices = []
        for i in range(1, 4):
            choice = random.randint(0, 4)
            compchoices.append(choice)
            print(symbols[choice])
            if i != 3:
                input("Ready for the next one?")
        if compchoices[0] == compchoices[1] and compchoices[1] == compchoices[2]:
            print("You WON!!")
            points = points + 1
        else:
            print("LOSER")
        user_ans = ask_again()
        if user_ans == "n":
            break    
    
    return points

def lucky_seven():
    print("The player rolls two dice and wins if the sum is 7.")
    points = 0
    while True:
        input("Are you ready for the first roll?")
        choice = random.randint(1, 6)
        print(f"You rolled a {choice}")
        input("Are you ready for the second roll?")
        choice2 = random.randint(1, 6)
        print(f"You rolled a {choice2}")
        total = choice + choice2
        if total == 7:
            print("You win!!")
            points = points + 1
        else:
            print("LOSER")
        print(f"Total: {total}")
        user_ans = ask_again()
        if user_ans == "n":
            break 

    return points

def blackjack():
    print("The player tries to get as close to 21 as possible without going over")
    points = 0
    while True:
        bet = int(input("How much do you bet?: "))
        total = 0
        i = 0
        stop = False
        card = random.randint(1, 10)
        print(f"Your first card is a {card}")
        total = total + card
        card = random.randint(1, 10)
        print(f"Your second card is a {card}")
        total = total + card
        print(f"Your current total is {total}")
        while stop == False:
            i = i + 1
            if i != 1:
                print(f"Your current total is {total}")
            ask = input("Hit or stand? (y/n): ")
            if ask == "y":
                card = random.randint(1, 10)
                total = card + total
                print(f"You got a {card}")
            elif ask == "n":
                print("Stopping now...")
                stop = True
            if total > 21:
                print("You lose!!")
                stop = True

        error = 21 - total
        print(f"Your total was {total}")
        print(f"You were {error} off")
        if error == 0:
            print("YOU WINN")
            points = points + bet
        elif error < 0:
            print("LOOOSEER")
            points = points - bet
        else:
            points = points + round(-error + (bet ** -2))
            print(round(-error + (bet ** -2)))
        user_ans = ask_again()

            break
